data['Class'] = numpy.floor(data['Distance']/3)
data['Class'] = data['Class'].astype(int)
data['Class'] = data['Class'].astype(str)

# lets find the frequency of each class
data2 = data.groupby('Class').count()
data2 = data2.reset_index()
data2.columns = ['Class', 'Frequency']

# lets find the min and max of each class and next to frequency
data3 = data.groupby('Class').agg({'Distance': ['min', 'max']})

# combine the two dataframes
